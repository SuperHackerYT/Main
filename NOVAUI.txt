local Utility = {}

function Utility.Create(className, properties, children)
    local instance = Instance.new(className)
    if properties then
        for prop, value in pairs(properties) do
            if prop ~= "Parent" then
                pcall(function()
                    instance[prop] = value
                end)
            end
        end
        if properties.Parent then
            instance.Parent = properties.Parent
        end
    end
    if children then
        for _, child in ipairs(children) do
            child.Parent = instance
        end
    end
    return instance
end

function Utility.Tween(instance, properties, duration, easingStyle, easingDirection)
    duration = duration or 0.3
    easingStyle = easingStyle or Enum.EasingStyle.Quart
    easingDirection = easingDirection or Enum.EasingDirection.Out
    local tween = TweenService:Create(
        instance,
        TweenInfo.new(duration, easingStyle, easingDirection),
        properties
    )
    tween:Play()
    return tween
end

function Utility.Ripple(button, x, y)
    local ripple = Utility.Create("Frame", {
        Name = "Ripple",
        Parent = button,
        BackgroundColor3 = Color3.fromRGB(255, 255, 255),
        BackgroundTransparency = 0.7,
        BorderSizePixel = 0,
        Position = UDim2.new(0, x - button.AbsolutePosition.X, 0, y - button.AbsolutePosition.Y),
        Size = UDim2.fromOffset(0, 0),
        AnchorPoint = Vector2.new(0.5, 0.5),
        ZIndex = button.ZIndex + 5
    })
    Utility.Create("UICorner", {
        CornerRadius = UDim.new(1, 0),
        Parent = ripple
    })
    local maxSize = math.max(button.AbsoluteSize.X, button.AbsoluteSize.Y) * 2.5
    local tween = Utility.Tween(ripple, {
        Size = UDim2.fromOffset(maxSize, maxSize),
        BackgroundTransparency = 1
    }, 0.5, Enum.EasingStyle.Quart, Enum.EasingDirection.Out)
    tween.Completed:Connect(function()
        ripple:Destroy()
    end)
end

function Utility.DeepCopy(original)
    local copy = {}
    for key, value in pairs(original) do
        if type(value) == "table" then
            copy[key] = Utility.DeepCopy(value)
        else
            copy[key] = value
        end
    end
    return copy
end

function Utility.Truncate(text, maxLen)
    if #text > maxLen then
        return string.sub(text, 1, maxLen - 3) .. "..."
    end
    return text
end

function Utility.IsMobile()
    return UserInputService.TouchEnabled and not UserInputService.KeyboardEnabled
end

function Utility.GetTextSize(text, fontSize, font, bounds)
    bounds = bounds or Vector2.new(math.huge, math.huge)
    local params = Instance.new("GetTextBoundsParams")
    params.Text = text
    params.Size = fontSize
    params.Font = font or Font.new("rbxasset://fonts/families/GothamSSm.json")
    params.Width = bounds.X
    local success, result = pcall(function()
        return TextService:GetTextBoundsAsync(params)
    end)
    if success then
        return result
    end
    return Vector2.new(#text * fontSize * 0.5, fontSize * 1.2)
end

function Utility.Color3ToHex(color)
    return string.format("#%02X%02X%02X",
        math.floor(color.R * 255),
        math.floor(color.G * 255),
        math.floor(color.B * 255)
    )
end

function Utility.HexToColor3(hex)
    hex = hex:gsub("#", "")
    return Color3.fromRGB(
        tonumber(hex:sub(1, 2), 16),
        tonumber(hex:sub(3, 4), 16),
        tonumber(hex:sub(5, 6), 16)
    )
end

function Utility.Lerp(a, b, t)
    return a + (b - a) * t
end

function Utility.LerpColor(c1, c2, t)
    return Color3.new(
        Utility.Lerp(c1.R, c2.R, t),
        Utility.Lerp(c1.G, c2.G, t),
        Utility.Lerp(c1.B, c2.B, t)
    )
end

function IconManager.GetIcon(name)
    if not name then return nil end
    if string.match(name, "^rbxassetid://") then
        return name
    end
    return IconManager.Icons[string.lower(name)] or IconManager.Icons["star"]
end

function IconManager.CreateIcon(name, size, parent, properties)
    size = size or 16
    local iconId = IconManager.GetIcon(name)
    if not iconId then return nil end
    
    local props = {
        Name = "Icon",
        Size = UDim2.fromOffset(size, size),
        BackgroundTransparency = 1,
        Image = iconId,
        ImageColor3 = Color3.fromRGB(255, 255, 255),
        ScaleType = Enum.ScaleType.Fit,
        Parent = parent
    }
    if properties then
        for k, v in pairs(properties) do
            props[k] = v
        end
    end
    return Utility.Create("ImageLabel", props)
end

function ThemeManager:Get(property)
    local theme = self.Themes[self.CurrentTheme]
    return theme and theme[property]
end

function ThemeManager:SetTheme(themeName)
    if not self.Themes[themeName] then return end
    self.CurrentTheme = themeName
    for _, listener in pairs(self.Listeners) do
        task.spawn(listener, themeName)
    end
end

function ThemeManager:OnThemeChanged(callback)
    self.ListenerId += 1
    local id = self.ListenerId
    self.Listeners[id] = callback
    return function()
        self.Listeners[id] = nil
    end
end

function ThemeManager:RegisterTheme(name, themeData)
    self.Themes[name] = themeData
end

function ThemeManager:GetThemeNames()
    local names = {}
    for name in pairs(self.Themes) do
        table.insert(names, name)
    end
    return names
end

local AnimationManager = {}
AnimationManager.ActiveTweens = {}

function AnimationManager:Animate(instance, properties, duration, style, direction, callback)
    duration = duration or 0.3
    style = style or Enum.EasingStyle.Quart
    direction = direction or Enum.EasingDirection.Out
    
    local key = tostring(instance) .. tostring(properties)
    if self.ActiveTweens[key] then
        self.ActiveTweens[key]:Cancel()
    end
    
    local tween = TweenService:Create(instance, TweenInfo.new(duration, style, direction), properties)
    self.ActiveTweens[key] = tween
    tween:Play()
    
    if callback then
        tween.Completed:Connect(function()
            self.ActiveTweens[key] = nil
            callback()
        end)
    else
        tween.Completed:Connect(function()
            self.ActiveTweens[key] = nil
        end)
    end
    
    return tween
end

function AnimationManager:Spring(instance, property, target, stiffness, damping)
    stiffness = stiffness or 180
    damping = damping or 12
    local current = instance[property]
    local velocity = 0
    
    local conn
    conn = RunService.Heartbeat:Connect(function(dt)
        dt = math.min(dt, 1/30)
        local displacement = current - target
        local springForce = -stiffness * displacement
        local dampingForce = -damping * velocity
        velocity = velocity + (springForce + dampingForce) * dt
        current = current + velocity * dt
        
        if math.abs(velocity) < 0.01 and math.abs(current - target) < 0.01 then
            instance[property] = target
            conn:Disconnect()
            return
        end
        
        instance[property] = current
    end)
    
    return conn
end

function AnimationManager:FadeIn(instance, duration)
    instance.Visible = true
    if instance:IsA("GuiObject") then
        instance.BackgroundTransparency = 1
        return self:Animate(instance, {BackgroundTransparency = 0}, duration or 0.25)
    end
end

function AnimationManager:FadeOut(instance, duration, callback)
    if instance:IsA("GuiObject") then
        return self:Animate(instance, {BackgroundTransparency = 1}, duration or 0.25, nil, nil, function()
            instance.Visible = false
            if callback then callback() end
        end)
    end
end

function AnimationManager:SlideIn(instance, direction, duration)
    direction = direction or "Right"
    duration = duration or 0.35
    instance.Visible = true
    
    local originalPos = instance.Position
    local startPos
    
    if direction == "Right" then
        startPos = originalPos + UDim2.fromOffset(40, 0)
    elseif direction == "Left" then
        startPos = originalPos - UDim2.fromOffset(40, 0)
    elseif direction == "Up" then
        startPos = originalPos - UDim2.fromOffset(0, 40)
    elseif direction == "Down" then
        startPos = originalPos + UDim2.fromOffset(0, 40)
    end
    
    instance.Position = startPos
    if instance:IsA("GuiObject") then
        instance.BackgroundTransparency = math.min(instance.BackgroundTransparency + 0.3, 1)
    end
    
    return self:Animate(instance, {Position = originalPos}, duration, Enum.EasingStyle.Quart, Enum.EasingDirection.Out)
end

function AnimationManager:ScaleIn(instance, duration)

local ConfigManager = {}
ConfigManager.__index = ConfigManager
ConfigManager.Flags = {}
ConfigManager.AutoSaveEnabled = false
ConfigManager.AutoSaveInterval = 30
ConfigManager.SaveFolder = "NovaUI"
ConfigManager.ConfigName = "default"
ConfigManager.AutoSaveConnection = nil

function ConfigManager:SetFlag(flag, value)
    self.Flags[flag] = value
    if self.AutoSaveEnabled then
        task.defer(function()
            self:Save()
        end)
    end
end

function ConfigManager:GetFlag(flag, default)
    if self.Flags[flag] ~= nil then
        return self.Flags[flag]
    end
    return default
end

function ConfigManager:Save(name)
    name = name or self.ConfigName
    local success, err = pcall(function()
        if not isfolder(self.SaveFolder) then
            makefolder(self.SaveFolder)
        end
        
        local data = {}
        for flag, value in pairs(self.Flags) do
            if typeof(value) == "Color3" then
                data[flag] = {Type = "Color3", Value = Utility.Color3ToHex(value)}
            elseif typeof(value) == "UDim2" then
                data[flag] = {
                    Type = "UDim2",
                    Value = {value.X.Scale, value.X.Offset, value.Y.Scale, value.Y.Offset}
                }
            elseif typeof(value) == "EnumItem" then
                data[flag] = {Type = "EnumItem", Value = tostring(value)}
            elseif type(value) == "table" then
                data[flag] = {Type = "table", Value = value}
            else
                data[flag] = {Type = type(value), Value = value}
            end
        end
        
        writefile(self.SaveFolder .. "/" .. name .. ".json", HttpService:JSONEncode(data))
    end)
    
    return success, err
end

function ConfigManager:Load(name)
    name = name or self.ConfigName
    local success, result = pcall(function()
        if isfile(self.SaveFolder .. "/" .. name .. ".json") then
            local raw = readfile(self.SaveFolder .. "/" .. name .. ".json")
            local data = HttpService:JSONDecode(raw)
            
            for flag, entry in pairs(data) do
                if entry.Type == "Color3" then
                    self.Flags[flag] = Utility.HexToColor3(entry.Value)
                elseif entry.Type == "UDim2" then
                    local v = entry.Value
                    self.Flags[flag] = UDim2.new(v[1], v[2], v[3], v[4])
                elseif entry.Type == "table" then
                    self.Flags[flag] = entry.Value
                else
                    self.Flags[flag] = entry.Value
                end
            end
            
            return true
        end
        return false
    end)
    
    return success and result
end

function ConfigManager:Delete(name)
    name = name or self.ConfigName
    pcall(function()
        if isfile(self.SaveFolder .. "/" .. name .. ".json") then
            delfile(self.SaveFolder .. "/" .. name .. ".json")
        end
    end)
end

function ConfigManager:GetConfigs()
    local configs = {}
    pcall(function()
        if isfolder(self.SaveFolder) then
            for _, file in ipairs(listfiles(self.SaveFolder)) do
                if string.match(file, "%.json$") then
                    local name = string.match(file, "([^/\\]+)%.json$")
                    if name then
                        table.insert(configs, name)
                    end
                end
            end
        end
    end)
    return configs
end

function ConfigManager:EnableAutoSave(interval)
    self.AutoSaveEnabled = true
    self.AutoSaveInterval = interval or 30
    
    if self.AutoSaveConnection then
        self.AutoSaveConnection:Disconnect()
    end
    
    self.AutoSaveConnection = task.spawn(function()
        while self.AutoSaveEnabled do
            task.wait(self.AutoSaveInterval)
            if self.AutoSaveEnabled then
                self:Save()
            end
        end
    end)
end

function ConfigManager:DisableAutoSave()
    self.AutoSaveEnabled = false
end

local InputManager = {}
InputManager.Keybinds = {}
InputManager.Connections = {}
InputManager.Enabled = true

function InputManager:Init()
    if self._initialized then return end
    self._initialized = true
    
    table.insert(self.Connections, UserInputService.InputBegan:Connect(function(input, processed)
        if not self.Enabled then return end
        if processed then return end
        
        for _, bind in pairs(self.Keybinds) do
            if bind.Enabled and input.KeyCode == bind.Key then
                task.spawn(bind.Callback, input)
            end
        end
    end))
end

function InputManager:BindKey(id, key, callback)
    self:Init()
    self.Keybinds[id] = {
        Key = key,
        Callback = callback,
        Enabled = true
    }
end

function InputManager:UnbindKey(id)
    self.Keybinds[id] = nil
end

function InputManager:SetKeyEnabled(id, enabled)
    if self.Keybinds[id] then
        self.Keybinds[id].Enabled = enabled
    end
end

function InputManager:Cleanup()
    for _, conn in ipairs(self.Connections) do
        conn:Disconnect()
    end
    self.Connections = {}
    self.Keybinds = {}
    self._initialized = false
end

-- ─────────── SLIDER ───────────
function Components.CreateSlider(parent, data, theme, zBase)
    zBase = zBase or 10
    local title = data.Title or "Slider"
    local desc = data.Desc or data.Description
    local icon = data.Icon
    local min = data.Min or 0
    local max = data.Max or 100
    local step = data.Step or 1
    local default = data.Default or min
    local suffix = data.Suffix or ""
    local callback = data.Callback or function() end
    local enabled = data.Enabled ~= false
    local visible = data.Visible ~= false
    local flag = data.Flag
    
    local value = math.clamp(default, min, max)
    local hasDesc = desc and desc ~= ""
    local dragging = false
    
    local height = hasDesc and 68 or 52
    
    local container = Utility.Create("Frame", {
        Name = "Slider_" .. title,
        Parent = parent,
        BackgroundColor3 = theme.Surface,
        Size = UDim2.new(1, 0, 0, height),
        Visible = visible,
        ZIndex = zBase
    })
    
    Utility.Create("UICorner", {
        CornerRadius = theme.CornerRadius,
        Parent = container
    })
    
    local contentOffset = 0
    if icon then
        contentOffset = 28
        IconManager.CreateIcon(icon, 16, container, {
            Position = UDim2.new(0, 12, 0, hasDesc and 14 or 8),
            ImageColor3 = theme.TextSecondary,
            ZIndex = zBase + 2
        })
    end
    
    Utility.Create("TextLabel", {
        Name = "Title",
        Parent = container,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 12 + contentOffset, 0, hasDesc and 8 or 4),
        Size = UDim2.new(0.6, -(12 + contentOffset), 0, 18),
        Text = title,
        TextColor3 = theme.Text,
        TextSize = theme.FontSizeBody,
        FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.Medium),
        TextXAlignment = Enum.TextXAlignment.Left,
        TextTruncate = Enum.TextTruncate.AtEnd,
        ZIndex = zBase + 2
    })
    
    local valueLabel = Utility.Create("TextLabel", {
        Name = "ValueLabel",
        Parent = container,
        BackgroundTransparency = 1,
        Position = UDim2.new(0.6, 0, 0, hasDesc and 8 or 4),
        Size = UDim2.new(0.4, -12, 0, 18),
        Text = tostring(value) .. suffix,
        TextColor3 = theme.Primary,
        TextSize = theme.FontSizeBody,
        FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.SemiBold),
        TextXAlignment = Enum.TextXAlignment.Right,
        ZIndex = zBase + 2
    })
    
    if hasDesc then
        Utility.Create("TextLabel", {
            Name = "Description",
            Parent = container,
            BackgroundTransparency = 1,
            Position = UDim2.new(0, 12 + contentOffset, 0, 26),
            Size = UDim2.new(1, -(24 + contentOffset), 0, 14),
            Text = desc,
            TextColor3 = theme.TextTertiary,
            TextSize = theme.FontSizeSmall,
            FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json"),
            TextXAlignment = Enum.TextXAlignment.Left,
            TextTruncate = Enum.TextTruncate.AtEnd,
            ZIndex = zBase + 2
        })
    end
    
    local sliderY = hasDesc and 46 or 30
    
    -- Track
    local track = Utility.Create("Frame", {
        Name = "Track",
        Parent = container,
        BackgroundColor3 = theme.SliderTrack,
        Size = UDim2.new(1, -24, 0, 4),
        Position = UDim2.new(0, 12, 0, sliderY),
        ZIndex = zBase + 2
    })
    
    Utility.Create("UICorner", {
        CornerRadius = UDim.new(1, 0),
        Parent = track
    })
    
    -- Fill
    local fill = Utility.Create("Frame", {
        Name = "Fill",
        Parent = track,
        BackgroundColor3 = theme.SliderFill,
        Size = UDim2.new((value - min) / (max - min), 0, 1, 0),
        ZIndex = zBase + 3
    })
    
    Utility.Create("UICorner", {
        CornerRadius = UDim.new(1, 0),
        Parent = fill
    })
    
    -- Knob
    local knob = Utility.Create("Frame", {
        Name = "Knob",
        Parent = track,
        BackgroundColor3 = theme.SliderKnob,
        Size = UDim2.fromOffset(14, 14),
        Position = UDim2.new((value - min) / (max - min), 0, 0.5, 0),
        AnchorPoint = Vector2.new(0.5, 0.5),
        ZIndex = zBase + 4
    })
    
    Utility.Create("UICorner", {
        CornerRadius = UDim.new(1, 0),
        Parent = knob
    })
    
    CreateShadow(knob, theme, 1, 3, 0.65)
    
    -- Interaction area
    local interactArea = Utility.Create("TextButton", {
        Name = "InteractArea",
        Parent = container,
        BackgroundTransparency = 1,
        Size = UDim2.new(1, 0, 0, 20),
        Position = UDim2.new(0, 0, 0, sliderY - 8),
        Text = "",
        ZIndex = zBase + 5
    })
    
    local function updateSlider(newValue, animate)
        newValue = math.clamp(newValue, min, max)
        -- Snap to step
        newValue = math.floor((newValue - min) / step + 0.5) * step + min
        newValue = math.clamp(newValue, min, max)
        
        value = newValue
        local percent = (value - min) / (max - min)
        
        valueLabel.Text = tostring(value) .. suffix
        
        if animate ~= false then
            Utility.Tween(fill, {Size = UDim2.new(percent, 0, 1, 0)}, 0.1, Enum.EasingStyle.Quad)
            Utility.Tween(knob, {Position = UDim2.new(percent, 0, 0.5, 0)}, 0.1, Enum.EasingStyle.Quad)
        else
            fill.Size = UDim2.new(percent, 0, 1, 0)
            knob.Position = UDim2.new(percent, 0, 0.5, 0)
        end
    end
    
    local function handleInput()
        local trackPos = track.AbsolutePosition.X
        local trackWidth = track.AbsoluteSize.X
        local mouseX = math.clamp(Mouse.X, trackPos, trackPos + trackWidth)
        local percent = (mouseX - trackPos) / trackWidth
        local newValue = min + (max - min) * percent
        updateSlider(newValue)
        if flag then
            ConfigManager:SetFlag(flag, value)
        end
        task.spawn(callback, value)
    end
    
    interactArea.MouseButton1Down:Connect(function()
        if not enabled then return end
        dragging = true
        handleInput()
        
        -- Scale up knob
        Utility.Tween(knob, {Size = UDim2.fromOffset(18, 18)}, 0.15, Enum.EasingStyle.Back)
    end)
    
    local moveConn, releaseConn
    moveConn = UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or
                         input.UserInputType == Enum.UserInputType.Touch) then
            handleInput()
        end
    end)
    
    releaseConn = UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            if dragging then
                dragging = false
                Utility.Tween(knob, {Size = UDim2.fromOffset(14, 14)}, 0.15)
            end
        end
    end)
    
    -- Hover effect
    container.MouseEnter:Connect(function()
        if not enabled then return end
        Utility.Tween(container, {BackgroundColor3 = theme.SurfaceHover}, 0.15)
    end)
    
    container.MouseLeave:Connect(function()
        Utility.Tween(container, {BackgroundColor3 = theme.Surface}, 0.15)
    end)
    
    -- Set initial flag
    if flag then
        local saved = ConfigManager:GetFlag(flag)
        if saved ~= nil then
            value = saved
            updateSlider(value, false)
        else
            ConfigManager:SetFlag(flag, value)
        end
    end
    
    local api = {}
    
    function api:GetValue()
        return value
    end
    
    function api:SetValue(v)
        updateSlider(v)
        if flag then ConfigManager:SetFlag(flag, value) end
        task.spawn(callback, value)
    end
    
    function api:SetEnabled(s)
        enabled = s
        container.BackgroundTransparency = s and 0 or 0.5
    end
    
    function api:SetVisible(s)
        container.Visible = s
    end
    
    function api:SetTitle(t)
        container.Title.Text = t
    end
    
    function api:SetCallback(fn)
        callback = fn
    end
    
    function api:Destroy()
        if moveConn then moveConn:Disconnect() end
        if releaseConn then releaseConn:Disconnect() end
        container:Destroy()
    end
    
    return api
end

    local function isSelected(opt)
        if multi then
            for _, v in ipairs(selected) do
                if v == opt then return true end
            end
            return false
        end
        return selected == opt
    end
    
    local function buildOptions(filterText)
        -- Clear old
        for _, btn in ipairs(optionButtons) do
            btn:Destroy()
        end
        optionButtons = {}
        
        for i, opt in ipairs(options) do
            if filterText and filterText ~= "" then
                if not string.find(string.lower(opt), string.lower(filterText)) then
                    continue
                end
            end
            
            local optBtn = Utility.Create("TextButton", {
                Name = "Option_" .. opt,
                Parent = optionsFrame,
                BackgroundColor3 = isSelected(opt) and theme.DropdownSelected or theme.DropdownBackground,
                Size = UDim2.new(1, 0, 0, 28),
                Text = "",
                LayoutOrder = i,
                ZIndex = zBase + 3

                Parent = optBtn

                TextXAlignment = Enum.TextXAlignment.Left,
                ZIndex = zBase + 4
            })

                IconManager.CreateIcon("check", 12, optBtn, {
                    Position = UDim2.new(1, -20, 0.5, 0),
                    AnchorPoint = Vector2.new(0, 0.5),
                    ImageColor3 = theme.Primary,
                    ZIndex = zBase + 4
                })
            end

            end)
            
            optBtn.MouseLeave:Connect(function()

                    BackgroundColor3 = isSelected(opt) and theme.DropdownSelected or theme.DropdownBackground
                }, 0.1)
            end)
            
            optBtn.MouseButton1Click:Connect(function()
                if multi then
                    local idx = table.find(selected, opt)
                    if idx then
                        table.remove(selected, idx)
                    else
                        table.insert(selected, opt)
                    end
                    selectedLabel.Text = getDisplayText()
                    buildOptions(searchBox and searchBox.Text or nil)
                    if flag then ConfigManager:SetFlag(flag, selected) end
                    task.spawn(callback, selected)
                else
                    selected = opt
                    selectedLabel.Text = getDisplayText()
                    if flag then ConfigManager:SetFlag(flag, selected) end
                    task.spawn(callback, selected)
                    -- Close dropdown
                    isOpen = false
                    local totalHeight = headerHeight
                    Utility.Tween(container, {Size = UDim2.new(1, 0, 0, totalHeight)}, 0.25, Enum.EasingStyle.Quart)
                    Utility.Tween(chevron, {Rotation = 0}, 0.25)
                end
            end)
            
            table.insert(optionButtons, optBtn)
        end
    end
    
    if searchBox then
        searchBox:GetPropertyChangedSignal("Text"):Connect(function()
            buildOptions(searchBox.Text)
        end)
    end
    
    local function toggleDropdown()
        if not enabled then return end
        isOpen = not isOpen
        
        if isOpen then
            buildOptions()
            task.wait()
            local optionsHeight = optionsLayout.AbsoluteContentSize.Y
            local totalHeight = headerHeight + optionsHeight + 12
            totalHeight = math.min(totalHeight, headerHeight + 200)
            Utility.Tween(container, {Size = UDim2.new(1, 0, 0, totalHeight)}, 0.3, Enum.EasingStyle.Quart)
            Utility.Tween(chevron, {Rotation = 180}, 0.3)
        else
            Utility.Tween(container, {Size = UDim2.new(1, 0, 0, headerHeight)}, 0.25, Enum.EasingStyle.Quart)
            Utility.Tween(chevron, {Rotation = 0}, 0.25)
        end
    end
    
    headerBtn.MouseButton1Click:Connect(toggleDropdown)
    
    headerBtn.MouseEnter:Connect(function()
        if not enabled then return end
        Utility.Tween(container, {BackgroundColor3 = theme.SurfaceHover}, 0.15)
    end)
    
    headerBtn.MouseLeave:Connect(function()
        Utility.Tween(container, {BackgroundColor3 = theme.Surface}, 0.15)
    end)
    
    if flag then
        local saved = ConfigManager:GetFlag(flag)
        if saved ~= nil then
            selected = saved
            selectedLabel.Text = getDisplayText()
        else
            ConfigManager:SetFlag(flag, selected)
        end
    end
    
    local api = {}
    
    function api:GetValue()
        return selected
    end
    
    function api:SetValue(v)
        selected = v
        selectedLabel.Text = getDisplayText()
        if flag then ConfigManager:SetFlag(flag, selected) end
        task.spawn(callback, selected)
    end
    
    function api:SetOptions(newOptions)
        options = newOptions
        if isOpen then
            buildOptions()
        end
    end
    
    function api:SetEnabled(s)
        enabled = s
        container.BackgroundTransparency = s and 0 or 0.5
    end
    
    function api:SetVisible(s)
        container.Visible = s
    end
    
    function api:SetTitle(t)
        container.Title.Text = t
    end
    
    function api:SetCallback(fn)
        callback = fn
    end
    
    function api:Destroy()
        container:Destroy()
    end
    
    return api
end

-- ─────────── KEYBIND ───────────
function Components.CreateKeybind(parent, data, theme, zBase)
    zBase = zBase or 10
    local title = data.Title or "Keybind"
    local desc = data.Desc or data.Description
    local icon = data.Icon
    local default = data.Default or Enum.KeyCode.Unknown
    local callback = data.Callback or function() end
    local onPress = data.OnPress or function() end
    local enabled = data.Enabled ~= false
    local visible = data.Visible ~= false
    local flag = data.Flag
    
    local currentKey = default
    local listening = false
    local hasDesc = desc and desc ~= ""
    local height = hasDesc and 52 or 36
    
    local container = Utility.Create("Frame", {
        Name = "Keybind_" .. title,
        Parent = parent,
        BackgroundColor3 = theme.Surface,
        Size = UDim2.new(1, 0, 0, height),
        Visible = visible,
        ZIndex = zBase
    })
    
    Utility.Create("UICorner", {
        CornerRadius = theme.CornerRadius,
        Parent = container
    })
    
    local contentOffset = 0
    if icon then
        contentOffset = 28
        IconManager.CreateIcon(icon, 16, container, {
            Position = UDim2.new(0, 12, 0.5, hasDesc and -6 or 0),
            AnchorPoint = Vector2.new(0, 0.5),
            ImageColor3 = theme.TextSecondary,
            ZIndex = zBase + 2
        })
    end
    
    Utility.Create("TextLabel", {
        Name = "Title",
        Parent = container,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 12 + contentOffset, 0, hasDesc and 8 or 0),
        Size = UDim2.new(0.6, -(12 + contentOffset), 0, hasDesc and 18 or height),
        Text = title,
        TextColor3 = theme.Text,
        TextSize = theme.FontSizeBody,
        FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.Medium),
        TextXAlignment = Enum.TextXAlignment.Left,
        TextTruncate = Enum.TextTruncate.AtEnd,
        ZIndex = zBase + 2
    })
    
    if hasDesc then
        Utility.Create("TextLabel", {
            Name = "Description",
            Parent = container,
            BackgroundTransparency = 1,
            Position = UDim2.new(0, 12 + contentOffset, 0, 28),
            Size = UDim2.new(0.6, -(12 + contentOffset), 0, 16),
            Text = desc,
            TextColor3 = theme.TextTertiary,
            TextSize = theme.FontSizeSmall,
            FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json"),
            TextXAlignment = Enum.TextXAlignment.Left,
            TextTruncate = Enum.TextTruncate.AtEnd,
            ZIndex = zBase + 2
        })
    end
    
    local function keyName(key)
        if key == Enum.KeyCode.Unknown then return "None" end
        local name = key.Name
        name = name:gsub("LeftShift", "LShift"):gsub("RightShift", "RShift")
        name = name:gsub("LeftControl", "LCtrl"):gsub("RightControl", "RCtrl")
        name = name:gsub("LeftAlt", "LAlt"):gsub("RightAlt", "RAlt")
        return name
    end
    
    local keybindBtn = Utility.Create("TextButton", {
        Name = "KeybindButton",
        Parent = container,
        BackgroundColor3 = theme.BackgroundTertiary,
        Size = UDim2.new(0, 0, 0, 24),
        AutomaticSize = Enum.AutomaticSize.X,
        Position = UDim2.new(1, -12, 0.5, 0),
        AnchorPoint = Vector2.new(1, 0.5),
        Text = "",
        ZIndex = zBase + 2
    })
    
    Utility.Create("UICorner", {
        CornerRadius = theme.CornerRadiusSmall,
        Parent = keybindBtn
    })
    
    Utility.Create("UIPadding", {
        Parent = keybindBtn,
        PaddingLeft = UDim.new(0, 10),
        PaddingRight = UDim.new(0, 10)
    })
    
    local keyLabel = Utility.Create("TextLabel", {
        Name = "KeyLabel",
        Parent = keybindBtn,
        BackgroundTransparency = 1,
        Size = UDim2.fromScale(1, 1),
        Text = keyName(currentKey),
        TextColor3 = theme.TextSecondary,
        TextSize = theme.FontSizeSmall,
        FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.SemiBold),
        ZIndex = zBase + 3
    })
    
    local inputConn
    
    keybindBtn.MouseButton1Click:Connect(function()
        if not enabled then return end
        if listening then return end
        listening = true
        keyLabel.Text = "..."
        Utility.Tween(keybindBtn, {BackgroundColor3 = theme.Primary}, 0.2)
        Utility.Tween(keyLabel, {TextColor3 = Color3.fromRGB(255, 255, 255)}, 0.2)
        
        inputConn = UserInputService.InputBegan:Connect(function(input, processed)
            if input.UserInputType == Enum.UserInputType.Keyboard then
                listening = false
                inputConn:Disconnect()
                
                if input.KeyCode == Enum.KeyCode.Escape then
                    currentKey = Enum.KeyCode.Unknown
                else
                    currentKey = input.KeyCode
                end
                
                keyLabel.Text = keyName(currentKey)
                Utility.Tween(keybindBtn, {BackgroundColor3 = theme.BackgroundTertiary}, 0.2)
                Utility.Tween(keyLabel, {TextColor3 = theme.TextSecondary}, 0.2)
                
                if flag then
                    ConfigManager:SetFlag(flag, currentKey.Name)
                end
                task.spawn(callback, currentKey)
            end
        end)
    end)
    
    -- Listen for key press
    InputManager:Init()
    local bindId = "Keybind_" .. title .. "_" .. tostring(math.random(10000, 99999))
    
    local keyPressConn = UserInputService.InputBegan:Connect(function(input, processed)
        if processed then return end
        if not enabled then return end
        if listening then return end
        if input.KeyCode == currentKey and currentKey ~= Enum.KeyCode.Unknown then
            task.spawn(onPress, currentKey)
        end
    end)
    
    container.MouseEnter:Connect(function()
        if not enabled then return end
        Utility.Tween(container, {BackgroundColor3 = theme.SurfaceHover}, 0.15)
    end)
    
    container.MouseLeave:Connect(function()
        Utility.Tween(container, {BackgroundColor3 = theme.Surface}, 0.15)
    end)
    
    if flag then
        local saved = ConfigManager:GetFlag(flag)
        if saved ~= nil then
            pcall(function()
                currentKey = Enum.KeyCode[saved]
                keyLabel.Text = keyName(currentKey)
            end)
        else
            ConfigManager:SetFlag(flag, currentKey.Name)
        end
    end
    
    local api = {}
    
    function api:GetValue()
        return currentKey
    end
    
    function api:SetValue(key)
        currentKey = key
        keyLabel.Text = keyName(currentKey)
        if flag then ConfigManager:SetFlag(flag, currentKey.Name) end
    end
    
    function api:SetEnabled(s)
        enabled = s
        container.BackgroundTransparency = s and 0 or 0.5
    end
    
    function api:SetVisible(s)
        container.Visible = s
    end
    
    function api:SetCallback(fn)
        callback = fn
    end
    
    function api:SetOnPress(fn)
        onPress = fn
    end
    
    function api:Destroy()
        if inputConn then inputConn:Disconnect() end
        keyPressConn:Disconnect()
        container:Destroy()
    end
    
    return api
end

        task.spawn(callback, currentColor)
    end
    
    -- SV interaction
    local svDragging = false
    
    svField.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            svDragging = true
        end
    end)
    
    local svMoveConn = UserInputService.InputChanged:Connect(function(input)
        if svDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or
                           input.UserInputType == Enum.UserInputType.Touch) then
            local pos = svField.AbsolutePosition
            local size = svField.AbsoluteSize
            s = math.clamp((Mouse.X - pos.X) / size.X, 0, 1)
            v = 1 - math.clamp((Mouse.Y - pos.Y) / size.Y, 0, 1)
            updateColor()
        end
    end)
    
    -- Hue interaction
    local hueDragging = false
    
    hueBar.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            hueDragging = true
        end
    end)
    
    local hueMoveConn = UserInputService.InputChanged:Connect(function(input)
        if hueDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or
                            input.UserInputType == Enum.UserInputType.Touch) then
            local pos = hueBar.AbsolutePosition
            local size = hueBar.AbsoluteSize
            h = math.clamp((Mouse.Y - pos.Y) / size.Y, 0, 1)
            updateColor()
        end
    end)
    
    local releaseConn = UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            svDragging = false
            hueDragging = false
        end
    end)
    
    hexInput.FocusLost:Connect(function()
        local success, color = pcall(Utility.HexToColor3, hexInput.Text)
        if success and color then
            h, s, v = Color3.toHSV(color)
            updateColor()
        else
            hexInput.Text = Utility.Color3ToHex(currentColor)
        end
    end)
    
    -- Toggle
    headerBtn.MouseButton1Click:Connect(function()
        if not enabled then return end
        isOpen = not isOpen
        local totalHeight = isOpen and (headerHeight + 160) or headerHeight
        Utility.Tween(container, {Size = UDim2.new(1, 0, 0, totalHeight)}, 0.3, Enum.EasingStyle.Quart)
    end)
    
    headerBtn.MouseEnter:Connect(function()
        if not enabled then return end
        Utility.Tween(container, {BackgroundColor3 = theme.SurfaceHover}, 0.15)
    end)
    
    headerBtn.MouseLeave:Connect(function()
        Utility.Tween(container, {BackgroundColor3 = theme.Surface}, 0.15)
    end)
    
    if flag then
        local saved = ConfigManager:GetFlag(flag)
        if saved ~= nil then
            local success, color = pcall(Utility.HexToColor3, saved)
            if success and color then
                currentColor = color
                h, s, v = Color3.toHSV(currentColor)
                preview.BackgroundColor3 = currentColor
                svField.BackgroundColor3 = Color3.fromHSV(h, 1, 1)
                svCursor.Position = UDim2.new(s, 0, 1 - v, 0)
                hueCursor.Position = UDim2.new(0.5, 0, h, 0)
                hexInput.Text = Utility.Color3ToHex(currentColor)
            end
        else
            ConfigManager:SetFlag(flag, Utility.Color3ToHex(currentColor))
        end
    end
    
    local api = {}
    
    function api:GetValue()
        return currentColor
    end
    
    function api:SetValue(color)
        currentColor = color
        h, s, v = Color3.toHSV(currentColor)
        updateColor()
    end
    
    function api:SetEnabled(state)
        enabled = state
    end
    
    function api:SetVisible(state)
        container.Visible = state
    end
    
    function api:SetCallback(fn)
        callback = fn
    end
    
    function api:Destroy()
        svMoveConn:Disconnect()
        hueMoveConn:Disconnect()
        releaseConn:Disconnect()
        container:Destroy()
    end
    
    return api
end

function Section:CreateBadge(data)

    local comp = Components.CreateBadge(self.Container, data, self.Theme, self.ZBase)

end

local Tab = {}
Tab.__index = Tab

function Tab.new(parent, data, theme, zBase)
    local self = setmetatable({}, Tab)
    self.Title = data.Title or "Tab"
    self.Icon = data.Icon
    self.Theme = theme
    self.ZBase = zBase or 10
    self.Sections = {}
    self.Visible = false
    
    self.Container = Utility.Create("ScrollingFrame", {
        Name = "Tab_" .. self.Title,
        Parent = parent,
        BackgroundTransparency = 1,
        Size = UDim2.fromScale(1, 1),
        Visible = false,
        ScrollBarThickness = 3,
        ScrollBarImageColor3 = theme.ScrollBar,
        ScrollBarImageTransparency = 0.3,
        CanvasSize = UDim2.new(0, 0, 0, 0),
        AutomaticCanvasSize = Enum.AutomaticSize.Y,
        ScrollingDirection = Enum.ScrollingDirection.Y,
        ElasticBehavior = Enum.ElasticBehavior.Always,
        ZIndex = self.ZBase,
        BorderSizePixel = 0
    })
    
    Utility.Create("UIPadding", {
        Parent = self.Container,
        PaddingTop = UDim.new(0, 12),
        PaddingBottom = UDim.new(0, 12),
        PaddingLeft = UDim.new(0, 16),
        PaddingRight = UDim.new(0, 16)
    })
    
    Utility.Create("UIListLayout", {
        Parent = self.Container,
        SortOrder = Enum.SortOrder.LayoutOrder,
        Padding = UDim.new(0, 12)
    })
    
    return self
end

function Tab:Show()
    self.Visible = true
    self.Container.Visible = true
    self.Container.CanvasPosition = Vector2.new(0, 0)
end

function Tab:Hide()
    self.Visible = false
    self.Container.Visible = false
end

function Tab:CreateSection(title)
    local section = Section.new(self.Container, title, self.Theme, self.ZBase)
    section.Container.LayoutOrder = #self.Sections + 1
    table.insert(self.Sections, section)
    return section
end

-- Shortcut methods (no section needed)
function Tab:CreateButton(data) return self:CreateSection(""):CreateButton(data) end
function Tab:CreateToggle(data) return self:CreateSection(""):CreateToggle(data) end
function Tab:CreateSlider(data) return self:CreateSection(""):CreateSlider(data) end
function Tab:CreateInput(data) return self:CreateSection(""):CreateInput(data) end
function Tab:CreateDropdown(data) return self:CreateSection(""):CreateDropdown(data) end
function Tab:CreateMultiDropdown(data) return self:CreateSection(""):CreateMultiDropdown(data) end
function Tab:CreateKeybind(data) return self:CreateSection(""):CreateKeybind(data) end
function Tab:CreateColorPicker(data) return self:CreateSection(""):CreateColorPicker(data) end
function Tab:CreateLabel(data) return self:CreateSection(""):CreateLabel(data) end
function Tab:CreateParagraph(data) return self:CreateSection(""):CreateParagraph(data) end
function Tab:CreateDivider(data) return self:CreateSection(""):CreateDivider(data) end
function Tab:CreateProgressBar(data) return self:CreateSection(""):CreateProgressBar(data) end

function Tab:Destroy()
    for _, section in ipairs(self.Sections) do
        section:Destroy()
    end
    self.Container:Destroy()
end

local Window = {}
Window.__index = Window

function Window.new(screenGui, data, theme)
    local self = setmetatable({}, Window)
    
    self.ScreenGui = screenGui
    self.Title = data.Title or "NovaUI"
    self.Icon = data.Icon
    self.Theme = theme
    self.Size = data.Size or UDim2.fromOffset(620, 470)
    self.Position = data.Position or UDim2.new(0.5, 0, 0.5, 0)
    self.MinSize = data.MinSize or Vector2.new(450, 350)
    self.Tabs = {}
    self.ActiveTab = nil
    self.Minimized = false
    self.Maximized = false
    self.Hidden = false
    self.ToggleKey = data.ToggleKey or Enum.KeyCode.RightShift
    self.Connections = {}
    self.ZBase = 10
    
    self:Build()
    
    return self
end

function Window:SetupDragging()
    local dragging = false
    local dragStart = nil
    local startPos = nil
    
    self.TitleBar.InputBegan:Connect(function(input)
        if self.Maximized then return end
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = self.Frame.Position
        end
    end)
    
    local dragConn = UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or
                          input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - dragStart
            self.Frame.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
        end
    end)
    
    local releaseConn = UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end)
    
    table.insert(self.Connections, dragConn)
    table.insert(self.Connections, releaseConn)
end

function Window:SetupResizing()
    local theme = self.Theme
    
    -- Resize handle (bottom-right corner)
    local resizeHandle = Utility.Create("TextButton", {
        Name = "ResizeHandle",
        Parent = self.Frame,
        BackgroundTransparency = 1,
        Size = UDim2.fromOffset(16, 16),
        Position = UDim2.new(1, -16, 1, -16),
        Text = "",
        ZIndex = self.ZBase + 20
    })
    
    -- Resize indicator dots
    for i = 0, 2 do
        for j = 0, 2 do
            if i + j >= 2 then
                Utility.Create("Frame", {
                    Parent = resizeHandle,
                    BackgroundColor3 = theme.TextTertiary,
                    BackgroundTransparency = 0.5,
                    Size = UDim2.fromOffset(2, 2),
                    Position = UDim2.fromOffset(4 + i * 4, 4 + j * 4),
                    BorderSizePixel = 0,
                    ZIndex = self.ZBase + 21
                })
            end
        end
    end
    
    local resizing = false
    local resizeStart = nil
    local startSize = nil
    
    resizeHandle.InputBegan:Connect(function(input)
        if self.Maximized or self.Minimized then return end
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            resizing = true
            resizeStart = input.Position
            startSize = self.Frame.Size
        end
    end)
    
    local resizeMoveConn = UserInputService.InputChanged:Connect(function(input)
        if resizing and (input.UserInputType == Enum.UserInputType.MouseMovement or
                          input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - resizeStart
            local newWidth = math.max(self.MinSize.X, startSize.X.Offset + delta.X)
            local newHeight = math.max(self.MinSize.Y, startSize.Y.Offset + delta.Y)
            self.Frame.Size = UDim2.fromOffset(newWidth, newHeight)
        end
    end)
    
    local resizeReleaseConn = UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            resizing = false
        end
    end)
    
    table.insert(self.Connections, resizeMoveConn)
    table.insert(self.Connections, resizeReleaseConn)
end

function Window:SetupToggleKey()
    local toggleConn = UserInputService.InputBegan:Connect(function(input, processed)
        if processed then return end
        if input.KeyCode == self.ToggleKey then
            self:ToggleVisibility()
        end
    end)
    table.insert(self.Connections, toggleConn)
end

function Window:ToggleMinimize()
    self.Minimized = not self.Minimized
    if self.Minimized then
        self._savedSize = self.Frame.Size
        Utility.Tween(self.Frame, {
            Size = UDim2.fromOffset(self.Frame.Size.X.Offset, 42)
        }, 0.3, Enum.EasingStyle.Quart)
        self.ContentArea.Visible = false
    else
        self.ContentArea.Visible = true
        Utility.Tween(self.Frame, {
            Size = self._savedSize or self.Size
        }, 0.3, Enum.EasingStyle.Quart)
    end
end

function Window:ToggleMaximize()
    self.Maximized = not self.Maximized
    if self.Maximized then
        self._savedSize = self.Frame.Size
        self._savedPosition = self.Frame.Position
        self._savedAnchor = self.Frame.AnchorPoint
        
        Utility.Tween(self.Frame, {
            Size = UDim2.fromScale(1, 1),
            Position = UDim2.fromScale(0.5, 0.5),
            AnchorPoint = Vector2.new(0.5, 0.5)
        }, 0.35, Enum.EasingStyle.Quart)
    else
        Utility.Tween(self.Frame, {
            Size = self._savedSize or self.Size,
            Position = self._savedPosition or self.Position,
            AnchorPoint = self._savedAnchor or Vector2.new(0.5, 0.5)
        }, 0.35, Enum.EasingStyle.Quart)
    end
end

function Window:ToggleVisibility()
    self.Hidden = not self.Hidden
    if self.Hidden then
        Utility.Tween(self.Frame, {
            Size = UDim2.fromOffset(
                self.Frame.Size.X.Offset * 0.95,
                self.Frame.Size.Y.Offset * 0.95
            ),
            BackgroundTransparency = 1
        }, 0.25, Enum.EasingStyle.Quart, Enum.EasingDirection.In)
        task.delay(0.25, function()
            self.Frame.Visible = false
        end)
    else
        self.Frame.Visible = true
        self.Frame.Size = UDim2.fromOffset(
            self.Size.X.Offset * 0.95,
            self.Size.Y.Offset * 0.95
        )
        Utility.Tween(self.Frame, {
            Size = self.Size,
            BackgroundTransparency = 0
        }, 0.3, Enum.EasingStyle.Back, Enum.EasingDirection.Out)
    end
end

function Window:Close()
    Utility.Tween(self.Frame, {
        Size = UDim2.fromOffset(
            self.Frame.Size.X.Offset * 0.92,
            self.Frame.Size.Y.Offset * 0.92
        ),
        BackgroundTransparency = 1
    }, 0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.In)
    
    task.delay(0.35, function()
        self:Destroy()
    end)
end

function Window:FilterTabs(query)
    query = string.lower(query or "")
    for _, tabData in ipairs(self.Tabs) do
        if query == "" then
            tabData.SidebarButton.Visible = true
        else
            local match = string.find(string.lower(tabData.Tab.Title), query) ~= nil
            tabData.SidebarButton.Visible = match
        end
    end
end

function Window:CreateTab(data)
    data = data or {}
    local theme = self.Theme
    
    local tab = Tab.new(self.TabContent, data, theme, self.ZBase)
    
    -- Create sidebar button
    local sidebarBtn = Utility.Create("TextButton", {
        Name = "TabBtn_" .. tab.Title,
        Parent = self.TabList,
        BackgroundColor3 = theme.SidebarActive,
        BackgroundTransparency = 1,
        Size = UDim2.new(1, 0, 0, 34),
        Text = "",
        LayoutOrder = #self.Tabs + 1,
        ZIndex = self.ZBase + 5
    })
    
    Utility.Create("UICorner", {
        CornerRadius = theme.CornerRadiusSmall,
        Parent = sidebarBtn
    })
    
    local btnOffset = 0
    if data.Icon then
        btnOffset = 24
        IconManager.CreateIcon(data.Icon, 16, sidebarBtn, {
            Name = "TabIcon",
            Position = UDim2.new(0, 10, 0.5, 0),
            AnchorPoint = Vector2.new(0, 0.5),
            ImageColor3 = theme.TextTertiary,
            ZIndex = self.ZBase + 6
        })
    end
    
    Utility.Create("TextLabel", {
        Name = "TabTitle",
        Parent = sidebarBtn,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 10 + btnOffset, 0, 0),
        Size = UDim2.new(1, -(14 + btnOffset), 1, 0),
        Text = tab.Title,
        TextColor3 = theme.TextSecondary,
        TextSize = theme.FontSizeBody,
        FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.Medium),
        TextXAlignment = Enum.TextXAlignment.Left,
        TextTruncate = Enum.TextTruncate.AtEnd,
        ZIndex = self.ZBase + 6
    })
    
    -- Active indicator
    local indicator = Utility.Create("Frame", {
        Name = "Indicator",
        Parent = sidebarBtn,
        BackgroundColor3 = theme.Primary,
        Size = UDim2.new(0, 3, 0, 0),
        Position = UDim2.new(0, 0, 0.5, 0),
        AnchorPoint = Vector2.new(0, 0.5),
        ZIndex = self.ZBase + 7
    })
    
    Utility.Create("UICorner", {
        CornerRadius = UDim.new(0, 2),
        Parent = indicator
    })
    
    local tabEntry = {
        Tab = tab,
        SidebarButton = sidebarBtn,
        Indicator = indicator
    }
    
    local function selectTab()
        -- Deselect all
        for _, entry in ipairs(self.Tabs) do
            entry.Tab:Hide()
            Utility.Tween(entry.SidebarButton, {BackgroundTransparency = 1}, 0.2)
            Utility.Tween(entry.Indicator, {Size = UDim2.new(0, 3, 0, 0)}, 0.2)
            if entry.SidebarButton:FindFirstChild("TabTitle") then
                Utility.Tween(entry.SidebarButton.TabTitle, {TextColor3 = theme.TextSecondary}, 0.2)
            end
            if entry.SidebarButton:FindFirstChild("TabIcon") then
                Utility.Tween(entry.SidebarButton.TabIcon, {ImageColor3 = theme.TextTertiary}, 0.2)
            end
        end
        
        -- Select this one
        tab:Show()
        self.ActiveTab = tab
        Utility.Tween(sidebarBtn, {BackgroundTransparency = 0.7}, 0.2)
        Utility.Tween(indicator, {Size = UDim2.new(0, 3, 0, 18)}, 0.25, Enum.EasingStyle.Back)
        if sidebarBtn:FindFirstChild("TabTitle") then
            Utility.Tween(sidebarBtn.TabTitle, {TextColor3 = theme.Text}, 0.2)
        end
        if sidebarBtn:FindFirstChild("TabIcon") then
            Utility.Tween(sidebarBtn.TabIcon, {ImageColor3 = theme.Primary}, 0.2)
        end
    end
    
    sidebarBtn.MouseButton1Click:Connect(selectTab)
    
    sidebarBtn.MouseEnter:Connect(function()
        if self.ActiveTab ~= tab then
            Utility.Tween(sidebarBtn, {BackgroundTransparency = 0.8}, 0.15)
        end
    end)
    
    sidebarBtn.MouseLeave:Connect(function()
        if self.ActiveTab ~= tab then
            Utility.Tween(sidebarBtn, {BackgroundTransparency = 1}, 0.15)
        end
    end)
    
    table.insert(self.Tabs, tabEntry)
    
    -- Auto-select first tab
    if #self.Tabs == 1 then
        selectTab()
    end
    
    return tab
end

function Window:SelectTab(tabTitle)
    for _, entry in ipairs(self.Tabs) do
        if entry.Tab.Title == tabTitle then
            entry.SidebarButton.MouseButton1Click:Fire()
            break
        end
    end
end

function Window:Destroy()
    for _, conn in ipairs(self.Connections) do
        conn:Disconnect()
    end
    for _, entry in ipairs(self.Tabs) do
        entry.Tab:Destroy()
    end
    if self.Frame then
        self.Frame:Destroy()
    end
end

-- NOVA UI MAIN LIBRARY
-- ═══════════════════════════════════════════════════════════
local NovaUI = {}
NovaUI.__index = NovaUI
NovaUI.Version = "1.0.0"
NovaUI.Windows = {}
NovaUI.ScreenGui = nil
NovaUI.Initialized = false

function NovaUI:Init()
    if self.Initialized then return end
    self.Initialized = true
    
    -- Create ScreenGui
    self.ScreenGui = Utility.Create("ScreenGui", {
        Name = "NovaUI_" .. tostring(math.random(100000, 999999)),
        ResetOnSpawn = false,
        ZIndexBehavior = Enum.ZIndexBehavior.Sibling,
        IgnoreGuiInset = true,
        DisplayOrder = 100
    })
    
    -- Try to parent to CoreGui, fallback to PlayerGui
    local success = pcall(function()
        self.ScreenGui.Parent = CoreGui
    end)
    if not success then
        pcall(function()
            if syn and syn.protect_gui then
                syn.protect_gui(self.ScreenGui)
            end
            self.ScreenGui.Parent = CoreGui
        end)
        if not self.ScreenGui.Parent then
            self.ScreenGui.Parent = LocalPlayer:WaitForChild("PlayerGui")
        end
    end
    
    -- Auto UI Scale for mobile
    if Utility.IsMobile() then
        local scale = Utility.Create("UIScale", {
            Parent = self.ScreenGui,
            Scale = math.min(Camera.ViewportSize.X / 700, 1)
        })
        
        Camera:GetPropertyChangedSignal("ViewportSize"):Connect(function()
            scale.Scale = math.min(Camera.ViewportSize.X / 700, 1)
        end)
    end
    
    -- Initialize systems
    NotificationManager:Init(self.ScreenGui)
    TooltipSystem:Init(self.ScreenGui)
    InputManager:Init()
end

function NovaUI:CreateWindow(data)
    self:Init()
    data = data or {}
    
    local themeName = data.Theme or ThemeManager.CurrentTheme
    if not ThemeManager.Themes[themeName] then
        themeName = "Dark"
    end
    ThemeManager.CurrentTheme = themeName
    local theme = ThemeManager.Themes[themeName]
    
    if data.SaveFolder then
        ConfigManager.SaveFolder = data.SaveFolder
    end
    if data.ConfigName then
        ConfigManager.ConfigName = data.ConfigName
    end
    
    local window = Window.new(self.ScreenGui, data, theme)
    table.insert(self.Windows, window)
    
    return window
end

function NovaUI:Notify(data)
    self:Init()
    return NotificationManager:Push(data)
end

function NovaUI:Dialog(data)
    self:Init()
    return DialogManager:Show(self.ScreenGui, data)
end

function NovaUI:Destroy()
    for _, window in ipairs(self.Windows) do
        window:Destroy()
    end
    self.Windows = {}
    
    AnimationManager:Cleanup()
    InputManager:Cleanup()
    ConfigManager:DisableAutoSave()
    
    if self.ScreenGui then
        self.ScreenGui:Destroy()
        self.ScreenGui = nil
    end
    
    self.Initialized = false
end
